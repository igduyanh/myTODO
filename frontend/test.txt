// Kiểm thử giao diện người dùng (UI) - sử dụng mock API để test hành vi UI
// Không phụ thuộc vào backend thật, chỉ test DOM và tương tác người dùng
describe('Kiểm thử giao diện TodoApp', () => {
  let mockTodos;

  beforeEach(() => {
    mockTodos = [];

    cy.intercept('GET', '/api/todos', (req) => {
      req.reply({ statusCode: 200, body: mockTodos });
    }).as('getTodos');

    cy.intercept('POST', '/api/todos', (req) => {
      const newTodo = {
        _id: `${Date.now()}`,
        title: req.body.title,
        completed: false,
        createdAt: new Date().toISOString(),
      };
      mockTodos.push(newTodo);
      req.reply({ statusCode: 201, body: newTodo });
    }).as('createTodo');

    cy.intercept('PATCH', '/api/todos/*/toggle', (req) => {
      const tokens = req.url.split('/');
      const id = tokens[tokens.length - 2];
      const todo = mockTodos.find((item) => item._id === id);
      if (!todo) return req.reply({ statusCode: 404, body: { error: 'Todo not found' } });
      todo.completed = !todo.completed;
      req.reply({ statusCode: 200, body: todo });
    }).as('toggleTodo');

    cy.intercept('PUT', '/api/todos/*', (req) => {
      const id = req.url.split('/').pop();
      const todo = mockTodos.find((item) => item._id === id);
      if (!todo) return req.reply({ statusCode: 404 });
      todo.title = req.body.title;
      todo.completed = req.body.completed;
      req.reply({ statusCode: 200, body: todo });
    }).as('saveTodo');

    cy.intercept('DELETE', '/api/todos/*', (req) => {
      const id = req.url.split('/').pop();
      mockTodos = mockTodos.filter((item) => item._id !== id);
      req.reply({ statusCode: 200, body: { success: true } });
    }).as('deleteTodo');

    cy.intercept('DELETE', '/api/todos', (req) => {
      mockTodos = mockTodos.filter((item) => !item.completed);
      req.reply({ statusCode: 200, body: { deletedCount: 1 } });
    }).as('clearCompleted');

    cy.visit('/');
    cy.wait('@getTodos');
  });

  const todos = ['Item 1', 'Item 2', 'Item 3'];

  it('Thêm nhiều todo, kiểm tra thứ tự và focus', () => {
    todos.forEach((item, index) => {
      cy.get('.header input').type(`${item}{enter}`);
      cy.wait(300);
      cy.get('.todo-list li').eq(index).should('contain.text', item);
    });

    cy.get('.todo-list li').should('have.length', todos.length);
  });

  it('Chỉnh sửa todo bằng nút Edit (Enter và Escape)', () => {
    cy.get('.header input').type('Item 1{enter}');
    cy.wait(300);

    // Nhấn nút Edit (không dùng dblclick nữa)
    cy.get('.todo-list li .btn-primary').click();
    cy.wait(300);

    cy.get('.todo-list li input.form-control')
      .should('exist')
      .and('be.visible')
      .and('have.focus')
      .type('{selectall}Edited Item{enter}');
    cy.wait(300);

    cy.get('.todo-list li').should('contain.text', 'Edited Item');

    // Test ESC để hủy edit
    cy.get('.todo-list li .btn-primary').click();
    cy.wait(300);

    cy.get('.todo-list li input.form-control').type('{selectall}Cancel Edit{esc}');
    cy.wait(300);

    cy.get('.todo-list li').should('contain.text', 'Edited Item');
  });

  it('Xóa todo', () => {
    cy.get('.header input').type('Item 1{enter}');
    cy.wait(300);

    cy.get('.todo-list li .btn-danger').click();
    cy.wait(300);

    cy.get('.todo-list li').should('have.length', 0);
  });

  it('Toggle từng todo và toggle tất cả', () => {
    todos.forEach((item) => {
      cy.get('.header input').type(`${item}{enter}`);
      cy.wait(200);
    });

    // Chuyển đổi trạng thái 1 item
    cy.get('.todo-list li input[type=checkbox]').first().check();
    cy.wait(300);
    cy.get('.todo-list li').first().should('have.class', 'list-group-item-success');

    // Toggle tất cả
    cy.get('#toggle-all').check();
    cy.wait(300);

    cy.get('.todo-list li input[type=checkbox]').each(($el) => {
      cy.wrap($el).should('be.checked');
    });

    cy.get('#toggle-all').uncheck();
    cy.wait(300);

    cy.get('.todo-list li input[type=checkbox]').each(($el) => {
      cy.wrap($el).should('not.be.checked');
    });
  });

  it('Lọc todo (All / Active / Completed)', () => {
    todos.forEach((item) => {
      cy.get('.header input').type(`${item}{enter}`);
    });
    cy.wait(300);

    cy.get('.todo-list li input[type=checkbox]').first().check();
    cy.wait(300);

    // Chế độ Active
    cy.contains('button', 'Active').click();
    cy.wait(300);
    cy.get('.todo-list li').should('have.length', 2);

    // Completed
    cy.contains('button', 'Completed').click();
    cy.wait(300);
    cy.get('.todo-list li').should('have.length', 1).and('contain.text', 'Item 1');

    // All
    cy.contains('button', 'All').click();
    cy.wait(300);
    cy.get('.todo-list li').should('have.length', 3);
  });

  it('Xóa tất cả todo đã hoàn thành (Clear completed)', () => {
    todos.forEach((item) => {
      cy.get('.header input').type(`${item}{enter}`);
    });
    cy.wait(300);

    cy.get('.todo-list li input[type=checkbox]').first().check();
    cy.wait(300);

    cy.contains('button', 'Clear completed').click();
    cy.wait(300);

    cy.get('.todo-list li').should('have.length', 2);
  });

  it('Kiểm tra class UI (completed, editing)', () => {
    cy.get('.header input').type('Item 1{enter}');
    cy.wait(300);

    // completed
    cy.get('.todo-list li input[type=checkbox]').check();
    cy.wait(300);
    cy.get('.todo-list li').should('have.class', 'list-group-item-success');

    // editing
    cy.get('.todo-list li .btn-primary').click();
    cy.wait(300);
    cy.get('.todo-list li').should('have.class', 'editing');
  });

  it('Kiểm tra thao tác bàn phím khi edit', () => {
    cy.get('.header input').type('Item 1{enter}');
    cy.wait(300);

    cy.get('.todo-list li .btn-primary').click();
    cy.wait(300);

    cy.get('.todo-list li input.form-control').should('have.focus').type('{selectall}New Keyboard Item{enter}');
    cy.wait(300);

    cy.get('.todo-list li').should('contain.text', 'New Keyboard Item');
  });

  it('Kiểm tra các nút Edit, Delete, Checkbox tồn tại', () => {
    cy.get('.header input').type('Item 1{enter}');
    cy.wait(300);

    cy.get('.todo-list li').within(() => {
      cy.get('input[type=checkbox]').should('exist');
      cy.get('.btn-primary').should('exist').and('contain.text', 'Edit');
      cy.get('.btn-danger').should('exist').and('contain.text', 'Delete');
    });
  });

  it('Kiểm tra số lượng item left hiển thị đúng', () => {
    cy.get('.header input').type('Item 1{enter}');
    cy.get('.header input').type('Item 2{enter}');
    cy.wait(300);

    // Ban đầu: 2 item active
    cy.contains('2 items left').should('exist');

    // Check 1 item → còn 1
    cy.get('.todo-list li input[type=checkbox]').first().check();
    cy.wait(300);

    cy.contains('1 item left').should('exist');

    // Check hết → còn 0
    cy.get('.todo-list li input[type=checkbox]').last().check();
    cy.wait(300);

    cy.contains('0 items left').should('exist');
  });
});
